/** getTool
  *
  */
[RoutineDef, getTool
,
	//[Note, "getTool"]
	
	[VarSet, $tool_path, [FileNameAsk, "ZTL(*.ztl)", , "Please select a tool to load..."]]
	
	[If, [StrLength,$tool_path ] > 0
	, // THEN LOAD TOOL
		
		[RoutineCall, getToolPath, $index, $tool_path ]
		
		[RoutineCall, enableLoadButton, $index]
		
		[RoutineCall, loadToolToViewport, $tool_path]
	]
	
, // PARAMS
	$index
]
