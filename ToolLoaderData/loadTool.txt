/** load Tool from \toollist.zvr
  *
  * @param integer index of tool in file to load, index 0 is last used tool
  *
  *
  */
[RoutineDef, loadTool
,
	[VarDef, escapedPath, ""]

	[RoutineCall, escapePath, [FileNameExtract, loadToolPath, 1 ]]
	
	[RoutineCall, loadToolPath, index ]
	
	[If, [FileExists, TOOLPATHLOADED]
	, // THEN
		[RoutineCall, loadToolToViewport, TOOLPATHLOADED]

		[Note, [StrMerge, "TOOL LOADED\n\nDIR: ", escapedPath, "\n\nTOOL: ", [FileNameExtract, TOOLPATHLOADED, 2 ] ] ]

	, // ELSE
		[VarSet, TOOLLIST([Var, index]), ""]

		[RoutineCall, disableButton, index]

		[Note, [StrMerge, "TOOL DOES NOT EXISTS\n\n", [FileNameExtract, escapedPath, 2+4]],, 0.5]
	]

, // PARAMS
	index
]
